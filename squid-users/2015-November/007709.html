just tries the following url-rewrite-program (perl)

<url-rewrite-program.pl>
#!/usr/bin/perl -wl
$ |= 1;  # don't buffer the output
while ( <> )
{
     unless( m,(\S+) (\S+)/(\S+) (\S+) (\S+), )
     {
         $uri = ''; next;
     }
     $uri = $1;
     ...
     $uri = "301:https://rsa-md5.ssl.hboeck.de/" if ( $uri =~ 
m/^https:\/\/ssl\.hboeck\.de\/(\S*)/ );
}
continue
{
     print "$uri";
}
exit;
</url-rewrite-program.pl>
