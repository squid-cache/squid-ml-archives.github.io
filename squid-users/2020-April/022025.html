    --prefix=/usr \
    --exec-prefix=/usr \
    --bindir=/usr/bin \
    --sbindir=/usr/sbin \
    --libdir=/usr/lib \
    --libexecdir=/usr/libexec/squid \
    --includedir=/usr/include \
    --mandir=/usr/share/man \
    --infodir=/usr/share/info \
    --datadir=/usr/share/squid \

    --with-pidfile=/var/run/squid.pid \
    --with-default-user=squid \
    --with-openssl \
    --enable-ssl \
    --enable-ssl-crtd
