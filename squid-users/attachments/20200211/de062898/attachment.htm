<p class="MsoNormal"><span lang="EN-US" style="font-size:10.0pt;font-family:"Courier New""># add diladele apt key<o:p></o:p></span></p>
<p class="MsoNormal"><span lang="EN-US" style="font-size:10.0pt;font-family:"Courier New"">wget -qO -

<p class="MsoNormal"><span lang="EN-US" style="font-size:10.0pt;font-family:"Courier New"">echo "deb
<a href="http://squid410.diladele.com/ubuntu/">http://squid410.diladele.com/ubuntu/</a> bionic main" > /etc/apt/sources.list.d/squid410.diladele.com.list<o:p></o:p></span></p>
<p class="MsoNormal"><span lang="EN-US" style="font-size:10.0pt;font-family:"Courier New""><o:p> </o:p></span></p>
